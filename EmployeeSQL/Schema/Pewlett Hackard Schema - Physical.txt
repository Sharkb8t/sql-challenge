# Physical Schema

employees
-
emp_no INT PK
emp_title_id VARCHAR(10) FK >- titles.title_id
birth_date DATE
first_name VARCHAR(50)
last_name VARCHAR(50)
gender CHAR(1)
hire_date DATE

departments
-
dept_no CHAR(4) PK
dept_name VARCHAR(50) UNIQUE

dept_emp
-
emp_no INT PK FK >- employees.emp_no
dept_no CHAR(4) PK FK >- departments.dept_no
from_date DATE
to_date DATE

dept_manager
-
dept_no CHAR(4) PK FK >- departments.dept_no
emp_no INT PK FK >- employees.emp_no
from_date DATE
to_date DATE

salaries
-
emp_no INT PK FK >- employees.emp_no
salary INT
from_date DATE
to_date DATE

titles
-
title_id VARCHAR(10) PK
title VARCHAR(50)
